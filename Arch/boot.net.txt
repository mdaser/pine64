# After modifying, run ./mkscr

# Set root partition to the second partition of boot device
part uuid ${devtype} ${devnum}:${bootpart} uuid

setenv bootargs console=${console} root=PARTUUID=${uuid} rw rootwait

setenv serveraddr 192.168.2.180

if dhcp ${kernel_addr_r} ${serveraddr}:Image-4.16.xx; then
  setenv fdtfile sun50i-a64-${board_name}.dtb
  if dhcp ${fdt_addr_r} ${serveraddr}:${fdtfile}; then
    if dhcp ${ramdisk_addr_r} ${serveraddr}:initramfs-linux-4.16.xx.img; then
      booti ${kernel_addr_r} ${ramdisk_addr_r}:${filesize} ${fdt_addr_r};
    else
      booti ${kernel_addr_r} - ${fdt_addr_r};
    fi;
  fi;
fi
